82static std::string trimAscii(const char* text) {
83 if (text == nullptr) return std::string();
84 const char* start = text;
85 while (*start != '\0' && isSpaceChar(*start)) ++start;
86 const char* end = start + strlen(start);
87 while (end > start && isSpaceChar(*(end - 1))) --end;
88 return std::string(start, static_cast<size_t>(end - start));
89}

104static bool parseUnsignedNumber(const std::string& text, uint8_t radix, uint32_t& out) {
105 if (text.empty()) return false;
106 uint32_t value = 0U;
107 for (size_t i = 0; i < text.size(); ++i) {
108 const unsigned char ch = static_cast<unsigned char>(text[i]);
109 uint8_t digit = 0xFFU;
110 if (ch >= '0' && ch <= '9') digit = static_cast<uint8_t>(ch - '0');
111 else if (ch >= 'A' && ch <= 'F') digit = static_cast<uint8_t>(10U + (ch - 'A'));
112 else if (ch >= 'a' && ch <= 'f') digit = static_cast<uint8_t>(10U + (ch - 'a'));
113 if (digit == 0xFFU || digit >= radix) return false;
114 value = static_cast<uint32_t>(value * radix + digit);
115 }
116 out = value;
117 return true;
118}

133static Error parseDeviceOnly(const char* text, DeviceAddress& out, const DeviceMeta** out_meta = nullptr) {
134 const std::string trimmed = trimAscii(text);
135 if (trimmed.empty()) return Error::InvalidArgument;
136 const std::string upper = upperAscii(trimmed);
137
138 std::string number_part;
139 const DeviceMeta* meta = findDeviceMeta(upper, number_part);
140 if (meta == nullptr) return Error::UnsupportedDevice;
141
142 uint32_t number = 0U;
143 if (!parseUnsignedNumber(number_part, meta->radix, number)) return Error::InvalidArgument;
144
145 out.code = meta->code;
146 out.number = number;
147 if (out_meta != nullptr) *out_meta = meta;
148 return Error::Ok;

292 SlmpClient& client,
293 const LongReadAccess& access,

295 LongTimerResult& out

344static Error readRandomMaps(
345 SlmpClient& client,
346 const std::vector<DeviceAddress>& word_devices,
347 const std::vector<DeviceAddress>& dword_devices,
348 std::unordered_map<uint64_t, uint16_t>& word_values,
349 std::unordered_map<uint64_t, uint32_t>& dword_values
350) {
351 size_t word_index = 0U;
352 size_t dword_index = 0U;
353 while (word_index < word_devices.size() || dword_index < dword_devices.size()) {
354 const size_t word_chunk = (word_devices.size() - word_index > 0xFFU) ? 0xFFU : (word_devices.size() - word_index);
355 const size_t dword_chunk = (dword_devices.size() - dword_index > 0xFFU) ? 0xFFU : (dword_devices.size() - dword_index);
356
357 std::vector<uint16_t> words(word_chunk);
358 std::vector<uint32_t> dwords(dword_chunk);
359 const Error err = client.readRandom(
360 word_chunk > 0U ? &word_devices[word_index] : nullptr,
361 word_chunk,
362 word_chunk > 0U ? words.data() : nullptr,
363 words.size(),
364 dword_chunk > 0U ? &dword_devices[dword_index] : nullptr,
365 dword_chunk,
366 dword_chunk > 0U ? dwords.data() : nullptr,
367 dwords.size()
368 );
369 if (err != Error::Ok) return err;
370
371 for (size_t i = 0; i < word_chunk; ++i)
372 word_values[deviceKey(word_devices[word_index + i])] = words[i];
373 for (size_t i = 0; i < dword_chunk; ++i)
374 dword_values[deviceKey(dword_devices[dword_index + i])] = dwords[i];
375
376 word_index += word_chunk;
377 dword_index += dword_chunk;
378 }
379 return Error::Ok;
380}

426Error parseAddressSpec(const char* address, AddressSpec& out) {
427 const std::string text = trimAscii(address);
428 if (text.empty()) return Error::InvalidArgument;
429
430 const size_t colon = text.find(':');
431 if (colon != std::string::npos) {
432 DeviceAddress device{};
433 const DeviceMeta* meta = nullptr;
434 const Error err = parseDeviceOnly(text.substr(0, colon).c_str(), device, &meta);
435 if (err != Error::Ok) return err;
436
437 ValueType type = ValueType::U16;
438 if (!parseTypeText(text.substr(colon + 1U), type)) return Error::InvalidArgument;
439 if (!normalizeRequestedType(*meta, type)) return Error::InvalidArgument;
440 if (!validateLongReadType(device.code, type)) return Error::InvalidArgument;
441
442 out.device = device;
443 out.type = type;
444 out.bit_index = -1;
445 out.explicit_type = true;
446 return Error::Ok;
447 }
448
449 const size_t dot = text.find('.');
450 if (dot != std::string::npos) {
451 DeviceAddress device{};
452 const DeviceMeta* meta = nullptr;
453 const Error err = parseDeviceOnly(text.substr(0, dot).c_str(), device, &meta);
454 if (err != Error::Ok) return err;
455 if (meta->bit_unit) return Error::InvalidArgument;
456
457 uint32_t bit_index = 0U;
458 if (!parseUnsignedNumber(text.substr(dot + 1U), 16U, bit_index) || bit_index > 15U)
459 return Error::InvalidArgument;
460
461 out.device = device;
462 out.type = ValueType::Bit;
463 out.bit_index = static_cast<int>(bit_index);
464 out.explicit_type = false;
465 return Error::Ok;
466 }
467
468 DeviceAddress device{};
469 const DeviceMeta* meta = nullptr;
470 const Error err = parseDeviceOnly(text.c_str(), device, &meta);
471 if (err != Error::Ok) return err;
472 out.device = device;
473 out.type = defaultValueType(*meta);
474 out.bit_index = -1;
475 out.explicit_type = false;
476 return Error::Ok;
477}

479Error readTyped(SlmpClient& client, const char* device, const char* dtype, Value& out) {
480 if (device == nullptr || dtype == nullptr) return Error::InvalidArgument;
481 const std::string address = trimAscii(device);
482 const std::string suffix = trimAscii(dtype);
483 if (address.empty() || suffix.empty()) return Error::InvalidArgument;
484
485 AddressSpec spec{};
486 const std::string merged = address + ":" + suffix;
487 Error err = parseAddressSpec(merged.c_str(), spec);
488 if (err != Error::Ok) return err;
489 if (spec.bit_index >= 0) return Error::InvalidArgument;
490
491 LongReadAccess long_read{};
492 if (getLongReadAccess(spec.device.code, long_read)) {
493 LongTimerResult raw{};
494 err = readLongLikePoint(client, long_read, spec.device.number, raw);
495 if (err != Error::Ok) return err;
496 out = decodeLongReadValue(raw, long_read, spec.type);

509 uint32_t raw = 0UL;

513 return Error::Ok;
514 }
515
516 uint16_t raw = 0U;

518 if (err != Error::Ok) return err;

568Error writeBitInWord(SlmpClient& client, const char* device, int bit_index, bool value) {
569 if (device == nullptr || bit_index < 0 || bit_index > 15) return Error::InvalidArgument;
570 DeviceAddress base{};
571 const DeviceMeta* meta = nullptr;
572 Error err = parseDeviceOnly(device, base, &meta);
573 if (err != Error::Ok) return err;
574 if (meta->bit_unit) return Error::InvalidArgument;
575
576 uint16_t word = 0U;
577 err = client.readOneWord(base, word);
578 if (err != Error::Ok) return err;
579 if (value) word = static_cast<uint16_t>(word | (1U << bit_index));
580 else word = static_cast<uint16_t>(word & ~(1U << bit_index));
581 return client.writeOneWord(base, word);
582}

611Error readWordsChunked(
612 SlmpClient& client,
613 const char* start,
614 uint16_t count,
615 std::vector<uint16_t>& out,
616 uint16_t max_per_request,
617 bool allow_split
618) {
619 if (start == nullptr || count == 0U || max_per_request == 0U) return Error::InvalidArgument;
620 DeviceAddress device{};
621 const DeviceMeta* meta = nullptr;
622 Error err = parseDeviceOnly(start, device, &meta);
623 if (err != Error::Ok || meta->bit_unit) return Error::InvalidArgument;
624 if (count > max_per_request && !allow_split) return Error::InvalidArgument;
625
626 out.clear();
627 out.reserve(count);
628 uint32_t offset = 0U;
629 uint16_t remaining = count;
630 while (remaining > 0U) {
631 const uint16_t chunk = (remaining > max_per_request) ? max_per_request : remaining;
632 std::vector<uint16_t> buffer(chunk);
633 DeviceAddress current = device;
634 current.number += offset;
635 err = client.readWords(current, chunk, buffer.data(), buffer.size());
636 if (err != Error::Ok) return err;
637 out.insert(out.end(), buffer.begin(), buffer.end());
638 remaining = static_cast<uint16_t>(remaining - chunk);
639 offset += chunk;
640 }
641 return Error::Ok;
642}

647 uint16_t count,
648 std::vector<uint32_t>& out,
649 uint16_t max_dwords_per_request,

662 uint16_t remaining = count;
663 while (remaining > 0U) {
664 const uint16_t chunk = (remaining > max_dwords_per_request) ? max_dwords_per_request : remaining;
665 std::vector<uint32_t> buffer(chunk);
666 DeviceAddress current = device;
667 current.number += offset;
668 err = client.readDWords(current, chunk, buffer.data(), buffer.size());
669 if (err != Error::Ok) return err;
670 out.insert(out.end(), buffer.begin(), buffer.end());
671 remaining = static_cast<uint16_t>(remaining - chunk);
672 offset += static_cast<uint32_t>(chunk) * 2U;
673 }
674 return Error::Ok;
675}

677Error compileReadPlan(const std::vector<std::string>& addresses, ReadPlan& out) {
678 out.entries.clear();
679 out.word_devices.clear();
680 out.dword_devices.clear();
681 out.entries.reserve(addresses.size());
682
683 for (size_t i = 0; i < addresses.size(); ++i) {
684 AddressSpec spec{};
685 Error err = parseAddressSpec(addresses[i].c_str(), spec);
686 if (err != Error::Ok) return err;
687
688 const DeviceMeta* meta = nullptr;
689 DeviceAddress verified{};
690 err = parseDeviceOnly(addresses[i].find('.') != std::string::npos
691 ? addresses[i].substr(0, addresses[i].find('.')).c_str()
692 : addresses[i].find(':') != std::string::npos
693 ? addresses[i].substr(0, addresses[i].find(':')).c_str()
694 : addresses[i].c_str(),
695 verified,
696 &meta);
697 if (err != Error::Ok) return err;
698
699 BatchKind kind = BatchKind::None;
700 LongReadAccess long_read{};
701 if (spec.bit_index >= 0) {
702 if (meta->batchable_word) {
703 kind = BatchKind::BitInWord;
704 if (!containsDevice(out.word_devices, spec.device)) out.word_devices.push_back(spec.device);
705 }
706 } else if (getLongReadAccess(spec.device.code, long_read)) {
707 kind = BatchKind::LongTimer;
708 } else if (!meta->bit_unit && meta->batchable_word) {
709 if (valueTypeUsesDword(spec.type)) {
710 kind = BatchKind::Dword;
711 if (!containsDevice(out.dword_devices, spec.device)) out.dword_devices.push_back(spec.device);
712 } else {
713 kind = BatchKind::Word;
714 if (!containsDevice(out.word_devices, spec.device)) out.word_devices.push_back(spec.device);
715 }
716 }
717
718 ReadPlanEntry entry;
719 entry.address = addresses[i];
720 entry.spec = spec;
721 entry.kind = kind;
722 out.entries.push_back(entry);
723 }
724 return Error::Ok;
725}

729 Error err = compileReadPlan(addresses, plan);

734Error readNamed(SlmpClient& client, const ReadPlan& plan, Snapshot& out) {
735 out.clear();
736 out.reserve(plan.entries.size());
737
738 std::unordered_map<uint64_t, uint16_t> word_values;
739 std::unordered_map<uint64_t, uint32_t> dword_values;
740 std::unordered_map<uint64_t, LongTimerResult> long_values;
741 Error err = readRandomMaps(client, plan.word_devices, plan.dword_devices, word_values, dword_values);
742 if (err != Error::Ok) return err;
743
744 for (size_t i = 0; i < plan.entries.size(); ++i) {
745 const ReadPlanEntry& entry = plan.entries[i];
746 NamedValue item;
747 item.address = entry.address;
748
749 switch (entry.kind) {
750 case BatchKind::Word:
751 item.value = decodeWordValue(word_values[deviceKey(entry.spec.device)], entry.spec.type);
752 break;
753 case BatchKind::BitInWord: {
754 const uint16_t word = word_values[deviceKey(entry.spec.device)];
755 item.value = Value::bitValue(((word >> entry.spec.bit_index) & 1U) != 0U);
756 break;
757 }
758 case BatchKind::Dword:
759 item.value = decodeDwordValue(dword_values[deviceKey(entry.spec.device)], entry.spec.type);
760 break;
761 case BatchKind::LongTimer: {
762 LongReadAccess long_read{};
763 if (!getLongReadAccess(entry.spec.device.code, long_read)) return Error::InvalidArgument;
764 const DeviceAddress base_device{long_read.base_code, entry.spec.device.number};
765 const uint64_t key = deviceKey(base_device);
766 auto found = long_values.find(key);
767 if (found == long_values.end()) {
768 LongTimerResult raw{};
769 err = readLongLikePoint(client, long_read, entry.spec.device.number, raw);
770 if (err != Error::Ok) return err;
771 found = long_values.emplace(key, raw).first;
772 }
773 item.value = decodeLongReadValue(found->second, long_read, entry.spec.type);
774 break;
775 }
776 case BatchKind::None:
777 default:
778 err = readTyped(client, entry.address.c_str(), item.value);
779 if (err != Error::Ok) return err;
780 break;
781 }
782
783 out.push_back(item);
784 }
785
786 return Error::Ok;
787}
